package main

import (
	"fmt"
	"os"
	"strconv"
)

func main() {
	n := os.Args[1]
	n_int, _ := strconv.Atoi(n)
	fmt.Println(Fibo(n_int))
}

func Fibo(n int) (r int) {
	if n <= 1 {
		return 1
	} else {
		return Fibo(n-2) + Fibo(n-1)
	}
}
