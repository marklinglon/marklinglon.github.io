```
package main

import "fmt"

func sum(nums []int) int {
    result := 0
    for _, num := range nums {
        result += num
    }
    return result
}

func main() {
    numbers := []int{1, 2, 3, 4, 5}
    total := sum(numbers)
    fmt.Printf("数组 %v 的和是 %d\n", numbers, total)
}
```
