## 冒泡排序
```
package main

import "fmt"

func bubbleSort(arr []int) {
    n := len(arr)
    for i := 0; i < n-1; i++ {
        for j := 0; j < n-i-1; j++ {
            if arr[j] > arr[j+1] {
                arr[j], arr[j+1] = arr[j+1], arr[j]
            }
        }
    }
}

func main() {
    arr := []int{64, 34, 25, 12, 22, 11, 90}
    fmt.Println("原始数组:", arr)
    bubbleSort(arr)
    fmt.Println("排序后数组:", arr)
}
```

## 快速排序
```
package main

import "fmt"

func quickSort(arr []int) {
    if len(arr) < 2 {
        return
    }

    pivot := arr[0]
    var left, right []int

    for _, v := range arr[1:] {
        if v <= pivot {
            left = append(left, v)
        } else {
            right = append(right, v)
        }
    }

    quickSort(left)
    quickSort(right)

    copy(arr, left)
    arr[len(left)] = pivot
    copy(arr[len(left)+1:], right)
}

func main() {
    arr := []int{64, 34, 25, 12, 22, 11, 90}
    fmt.Println("原始数组:", arr)
    quickSort(arr)
    fmt.Println("排序后数组:", arr)
}
```

## 选择排序
```
package main

import "fmt"

func selectionSort(arr []int) {
    n := len(arr)
    for i := 0; i < n-1; i++ {
        minIdx := i
        for j := i + 1; j < n; j++ {
            if arr[j] < arr[minIdx] {
                minIdx = j
            }
        }
        arr[i], arr[minIdx] = arr[minIdx], arr[i]
    }
}

func main() {
    arr := []int{64, 25, 12, 22, 11}
    fmt.Println("未排序前:", arr)
    selectionSort(arr)
    fmt.Println("排序后:", arr)
}
```
